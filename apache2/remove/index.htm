Using Ubuntu

sudo service apache2 stop
dpkg -l | grep apache2
sudo apt-get purge -y apache2 apache2-bin apache2-data
dpkg -l | grep apache2
ls -l /etc/apache2 /var/www/html/
sudo rm -rf /var/www/html /etc/apache2
Alternative 
sudo dpkg -r apache2 apache2-bin apache2-data
dpkg -l | grep apache2
